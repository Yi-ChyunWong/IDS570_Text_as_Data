بسم الله الرحمن الرحيم الحمد لله العالمين والصلاة والسلام سيد المرسلين سيدنا ومولانا محمد وعلى آله وصحبه صلاة وسلاما دائمين يوم الدين. وبعد سير الأولين صارت عبرة للآخرين يرى الإنسان العبر حصلت لغيره فيعتبر ويطالع حديث الأمم السالفة جرى فينزجر. فسبحان حديث الأولين عبرة لقوم آخرين العبر والحكايات تسمى ليلة وليلة الغرائب والأمثال. حكايات الملك شهريار وأخيه الملك شاه الزمان حكي والله أنه مضى قديم الزمان وسالف العصر والأوان ملك ملوك ساسان بجزائر الهند والصين صاحب جند وأعوان وخدم وحشم ولدان أحدهما كبير والآخر صغير وكانا بطلين وكان الكبير أفرس الصغير وقد ملك البلاد وحكم بالعدل العباد وأحبه أهل بلاده ومملكته وكان اسمه الملك شهريار وكان أخوه الصغير اسمه الملك شاه زمان وكان ملك سمرقند العجم ولم يزل الأمر مستقيما بلادهما وكل منهما مملكته حاكم عادل رعيته مدة سنة وهم غاية البسط والانشراح. يزالا الحالة اشتاق الكبير أخيه الصغير فأمر وزيره يسافر إليه ويحضر فأجابه بالسمع والطاعة وسافر وصل بالسلامة ودخل أخيه وبلغه السلام وأعلمه أخاه مشتاق إليه وقصده يزوره فأجابه بالسمع والطاعة وتجهز وأخرج خيامه وبغاله وخدمه وأعوانه وأقام وزيره حاكما بلاده وخرج طالبا بلاد أخيه. فلما نصف الليل تذكر حاجة نسيها قصره فرجع ودخل قصره فوجد زوجته راقدة فراشه معانقة عبدا أسود العبيد فلما اسودت الدنيا وجهه وقال نفسه: الأمر وقع وأنا فارقت المدينة فكيف حال العاهرة غبت أخي مدة أنه سل سيفه وضرب الاثنين فقتلهما الفراش ورجع وقته وساعته وسار وصل مدينة أخيه ففرح أخيه بقدومه خرج إليه ولاقاه وسلم ففرح غاية الفرح وزين المدينة وجلس معه يتحدث بانشراح فتذكر الملك شاه زمان أمر زوجته فحصل عنده غم زائد واصفر لونه وضعف جسمه فلما رآه أخوه الحالة نفسه بسبب مفارقته بلاده وملكه فترك سبيله ولم يسأل ذلك. أنه قال الأيام: أخي باطني جرح ولم يخبره زوجته فقال: إني أريد تسافر معي الصيد والقنص لعله ينشرح صدرك فأبى فسافر أخوه وحده الصيد. وكان قصر الملك شبابيك تطل بستان أخيه فنظروا بباب القصر فتح وخرج جارية وعشرون عبدا وامرأة أخيه تمشي بينهم وهي غاية الحسن والجمال وصلوا فسقية وخلعوا ثيابهم وجلسوا بعضهم بامرأة الملك قالت: مسعود فجاءها عبد أسود فعانقها وعانقته وواقعها وكذلك باقي العبيد فعلوا بالجواري ولم يزالوا بوس وعناق ونحو ولى النهار. فلما الملك فقال: والله بليتي أخف البلية وقد هان عنده القهر والغم وقال: أعظم جرى ولم يزل أكل وشرب. وبعد جاء أخوه السفر فسلما بعضهما ونظر الملك شهريار أخيه الملك شاه زمان وقد رد لونه واحمر وجهه وصار يأكل بشهية بعدما قليل الأكل فتعجب وقال: أخي كنت أراك مصفر الوجه والآن رد لونك فأخبرني بحالك فقال له: تغير لوني فأذكره واعف عني إخبارك برد لوني فقال له: أخبرني أولا بتغير لونك وضعفك أسمعه. فقال له: أخي إنك أرسلت وزيرك إلي يطلبني للحضور يديك جهزت حالي وقد بررت مدينتي أني تذكرت الخرزة أعطيتها قصري فرجعت فوجدت زوجتي معها عبد أسود نائم فراشي فقتلتهما وجئت وأنا متفكر الأمر فهذا سبب تغير لوني وضعفي وأما رد لوني فاعف عني أذكره لك. فلما سمع أخوه كلامه قال له: أقسمت بالله تخبرني بسبب رد لونك فأعاد رآه فقال شهريار لأخيه شاه زمان: اجعل أنك مسافر للصيد والقنص واختف عندي وأنت تشاهد وتحققه عيناك فنادى الملك ساعته بالسفر فخرجت العساكر والخيام ظاهر المدينة وخرج الملك أنه جلس الخيام وقال لغلمانه يدخل علي أنه تنكر وخرج مختفيا القصر أخوه وجلس الشباك المطل البستان ساعة الزمان بالجواري وسيدتهم دخلوا العبيد وفعلوا قال أخوه واستمروا العصر. فلما الملك شهريار الأمر طار عقله رأسه وقال لأخيه شاه زمان: قم نسافر حال سبيلنا وليس حاجة بالملك ننظر جرى لأحد مثلنا فيكون موتنا خير حياتنا فأجابه لذلك. أنهما خرجا باب سري القصر ولم يزالا مسافرين أياما وليالي وصلا شجرة وسط مرج عندها بجانب البحر المالح فشربا العين وجلسا يستريحان. فلما ساعة مضت النهار بالبحر هاج وطلع عمود أسود صاعد السماء قاصد المرجة. فلما رأيا خافا وطلعا أعلى الشجرة وكانت عالية وصارا ينظران يكون الخبر بجني طويل القامة عريض الهامة واسع الصدر رأسه صندوق فطلع البر وأتى الشجرة فوقها وجلس تحتها وفتح الصندوق وأخرج علبة فتحها فخرجت صبية بهية كأنها الشمس المضيئة قال الشاعر: أشـرقـت فـي الــدجـــى فـــلاح الـــنـــهـــار واســـتـــنـــارت بـــنـــورهـــا الأســـحـــار تـــســـجـــد الـــكـــائنـــات بـــين يديهـــــــا حـــين تـــبـــدو وتـــهـــتـــك الأســـتــــــار أومـــضـــت بـــروق حـــمـــــاهـــــــا هـــطـــلـــت بـــالـــمـــدامـــع الأمـــطــــار قال: فلما نظر إليها الجني قال: سيدة الحرائر اختطفتك ليلة عرسك أريد أنام قليلا الجني وضع رأسه ركبتيها ونام فرفعت رأسها أعلى الشجرة فرأت الملكين وهما الشجرة فرفعت رأس الجني ركبتيها ووضعته الأرض ووقفت الشجرة وقالت بالإشارة انزلا تخافا العفريت فقالا لها: بالله تسامحينا الأمر فقالت بالله عليكما تنزلا وإلا نبهت عليكما العفريت فيقتلكما شر قتلة فخافا ونزلا إليها فقامت وقالت ارصعا رصعا عنيفا وإلا أنبه عليكما العفريت خوفهما قال الملك شهريار لأخيه الملك شاه زمان: أخي افعل أمرتك فقال: أفعل تفعل قبلي وأخذا يتغامزان نكاحها فقالت أراكما تتغامزان تتقدما وتفعلا وإلا نبهت عليكما العفريت خوفهما الجني فعلا أمرتهما فلما فرغا قالت أقفا وأخرجت جيبها كيسا وأخرجت عقدا وسبعون خاتما فقالت لهما: أتدرون فقالا لها: ندري فقالت أصحاب الخواتم كلهم كانوا غفلة قرن العفريت فأعطياني خاتميكما الاثنان الأخران فأعطاها يديهما خاتمين فقالت العفريت اختطفني ليلة عرسي أنه وضعني علبة وجعل العلبة داخل الصندوق ورمى الصندوق أقفال وجعلني قاع البحر العجاج المتلاطم بالأمواج ويعلم المرأة منا أرادت أمر يغلبها شيء قال بعضهم: تـــأمـــنـــن إلــــى الـــــــنـــــــســـــــاء تـــثــــــق بـــــــعـــــــهـــــــودهـــــــن فـــرضـــاؤهـــن وســـخــــــطـــــــهـــــــن مـــعـــلــــــق بـــــــفـــــــروجـــــــهـــــــن يبدين ودا كـــــــــــــــاذبـــــــــــــــا والـــغـــدر حـــشــــو ثـــــــيابـــــــهـــــــن بـــحـــديث يوســـف فـــاعــــــتـــــــبـــــــرمـــتـــــحـــــــذرا مـــــــن كـــــــيدهـــــــن مـــــــا تـــــــرى إبــــــــلـــــــــــــــيس أخـــرج آدمـــا مــــن أجـــــــلـــــــهـــــــن فلما الكلام تعجبا غاية العجب وقالا لبعضهما: عفريتا وجرى أعظم جرى فهذا شيء يسلينا. أنهما انصرفا ساعتهما ورجعا مدينة الملك شهريار ودخلا قصره. أنه رمى عنق زوجته وكذلك أعناق الجواري والعبيد وصار الملك شهريار يأخذ بنتا بكرا يزيل بكارتها ويقتلها ليلتها ولم يزل مدة سنوات فضجت الناس وهربت ببناتها ولم يبق المدينة بنت تتحمل الوطء. الملك أمر الوزير يأتيه بنت جري عادته فخرج الوزير وفتش فلم يجد بنتا فتوجه منزله غضبان مقهور خائف نفسه الملك. وكان الوزير بنتان ذاتا حسن وجمال وبهاء وقد واعتدال الكبيرة اسمها شهرزاد والصغيرة اسمها دنيازاد وكانت الكبيرة قرأت الكتب والتواريخ وسير الملوك المتقدمين وأخبار الأمم الماضيين. قيل أنها جمعت كتاب كتب التواريخ المتعلقة بالأمم السالفة والملوك الخالية والشعراء فقالت لأبيها: مالي أراك متغيرا حامل الهم والأحزان وقد قال بعضهم المعنى شعرا: قـــل لـــمـــن يحــــــمـــــــل هـــــــمـــــــا هـــــــــــــــمـــــــــــــــــــــــــا يدوم مـــثـــل مـــا يفـــنـــــــى الـــــــســـــــرور هـــكـــذا تـــفـــنــــى الـــــــهـــــــمـــــــوم فلما سمع الوزير ابنته الكلام حكى جرى الأول الآخر الملك فقالت له: بالله أبت زوجني الملك فإما أعيش وإما أكون فداء لبنات المسلمين وسببا لخلاصهن يديه فقال لها: بالله تخاطري بنفسك فقالت له: بد فقال: أخشى يحصل حصل الحمار والثور صاحب الزرع فقالت له: جرى أبت حكاية الحمار والثور صاحب الزرع قال: اعلمي ابنتي أنه لبعض التجار أموال ومواش وكان زوجة وأولاد وكان الله تعالى أعطاه معرفة ألسن الحيوانات والطير وكان مسكن التاجر الأرياف وكان عنده داره حمار وثور فأتى يوما الثور مكان الحمار فوجده منكوسا مرشوشا وفي معلفه شعير مغربل وتبن مغربل راقد مستريح وفي الأوقات ركبه صاحبه لحاجة تعرض ويرجع حاله فلما الأيام سمع التاجر الثور يقول للحمار: هنيئا تعبان وأنت مستريح تأكل الشعير مغربلا ويخدمونك وفي الأوقات يركبك صاحبك ويرجع وأنا دائما للحرث. فقال الحمار: خرجت الغيط ووضعوا رقبتك الناف فارقد تقم ضربوك قمت فارقد ثانيا رجعوا ووضعوا الفول تأكله كأنك ضعيف وامتنع الأكل والشرب يوما يومين فإنك تستريح التعب والجهد وكان التاجر يسمع كلامهما فلما جاء السواق الثور بعلفه أكل شيئا يسيرا فأصبح السواق يأخذ الثور الحرث فوجده ضعيفا فقال التاجر: خذ الحمار وحرثه مكانه اليوم كله فلما آخر النهار شكره الثور تفضلاته أراحه التعب اليوم فلم يرد الحمار جوابا وندم أشد الندامة فلما يوم جاء المزارع وأخذ الحمار وحرثه آخر النهار فلم يرجع مسلوخ الرقبة شديد الضعف فتأمله الثور وشكره ومجده فقال الحمار: أني ناصح وقد سمعت صاحبنا يقول: يقم الثور موضعه فأعطوه للجزار ليذبحه ويعمل جلده قطعا وأنا خائف ونصحتك والسلام. فلما سمع الثور كلام الحمار شكره وقال غد أسرح معهم الثور أكل علفه بتمامه لحس المذود بلسانه وصاحبهما يسمع كلامهما فلما طلع النهار وخرج التاجر وزوجه دار البقر وجلسا فجاء السواق وأخذ الثور وخرج فلما الثور صاحبه حرك ذنبه وظرط وبرطع فضحك التاجر استلقى قفاه. فقالت زوجته: شيء تضحك فقال لها: شيء رأيته وسمعته أقدر أبيح فأموت فقالت له: بد تخبرني بذلك سبب ضحكك كنت تموت فقال لها: أقدر أبوح خوفا الموت فقالت له: تضحك علي. أنها تزل تلح وتلح الكلام غلبت فتحير أحضر أولاده وأرسل أحضر القاضي والشهود وأراد يوصي يبوح بالسر ويموت لأنه يحبها محبة عظيمة لأنها بنت عمه وأم أولاده وكان أنه أرسل وأحضر أهلها وأهل جارته وقال حكايته وأنه قال لأحد سره مات فقال الناس حضر: بالله اتركي الأمر لئلا يموت زوجك أولادك فقالت لهم: أرجع عنه يقول يموت. فسكتوا عنها. التاجر عندهم وتوجه دار الدواب ليتوضأ يرجع يقول ويموت. وكان عنده ديك تحته دجاجة وكان عنده كلب فسمع التاجر الكلب ينادي الديك ويسبه ويقول له: فرحان وصاحبنا رايح يموت فقال الديك للكلب: وكيف الأمر فأعاد الكلب القصة فقال الديك: والله صاحبنا قليل العقل. زوجة أرضي وأغضب زوجة واحدة يعرف صلاح أمره معها فما يأخذ بعضا عيدان التوت يدخل حجرتها ويضربها تموت تتوب تعود تسأله شيء. قال: فلما سمع التاجر كلام الديك يخاطب الكلب عقله وعزم ضربها قال الوزير لابنته شهرزاد ربما فعل فعل التاجر بزوجته فقالت له: فعل قال: دخل عليها الحجرة بعدما قطع عيدان التوت وخبأها داخل الحجرة وقال لها: تعالي داخل الحجرة أقول ينظرني أموت فدخلت معه أنه قفل باب الحجرة عليهما ونزل عليها بالضرب أغمي عليها فقالت له: تبت أنها قبلت يديه ورجليه وتابت وخرجت وإياه وفرح الجماعة وأهلها وقعدوا أسر الأحوال الممات. فلما سمعت ابنة الوزير مقالة أبيها قالت له: بد فجهزها وطلع الملك شهريار وكانت أوصت أختها الصغيرة وقالت لها: توجهت الملك أرسلت أطلبك جئت عندي ورأيت الملك قضى حاجته مني قولي أختي حدثينا حديثا غريبا نقطع السهر وأنا أحدثك حديثا يكون الخلاص شاء الله. أباها الوزير طلع الملك فلما رآه فرح وقال: أتيت بحاجتي فقال: فلما أراد يدخل عليها بكت فقال لها: فقالت: الملك أختا صغيرة أريد أودعها فأرسلها الملك إليها فجاءت أختها وعانقتها وجلست السرير فقام الملك وأخذ بكارتها جلسوا يتحدثون فقالت أختها الصغيرة: بالله أختي حدثينا حديثا نقطع سهر ليلتنا فقالت: حبا وكرامة أذن الملك المهذب فلما سمع الكلام وكان قلق ففرح بسماع الحديث. حكاية التاجر العفريت
